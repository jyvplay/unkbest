import { channel } from 'node:diagnostics_channel';
import { monitorEventLoopDelay } from 'node:perf_hooks';

export type RuntimePressure = 'ok' | 'warm' | 'hot';

export interface RuntimeEvent {
  ts: number;
  iso: string;
  type: string;
  payload?: unknown;
}

export interface RuntimeStats {
  pressure: RuntimePressure;
  eventLoop: {
    meanMs: number;
    maxMs: number;
    minMs: number;
    stddevMs: number;
  };
  memory: {
    rss: number;
    heapUsed: number;
    heapTotal: number;
    external: number;
    arrayBuffers: number;
  };
  uptimeSec: number;
  recentEventCount: number;
}

const runtimeChannel = channel('vite-native-scraper.runtime');
const lag = monitorEventLoopDelay({ resolution: 20 });
lag.enable();

const MAX_EVENTS = 500;
const recentEvents: RuntimeEvent[] = [];

const lagResetTimer = setInterval(() => {
  try { lag.reset(); } catch {}
}, 60_000);
(lagResetTimer as any).unref?.();

export function emitRuntimeEvent(type: string, payload?: unknown): void {
  const ev: RuntimeEvent = {
    ts: Date.now(),
    iso: new Date().toISOString(),
    type,
    payload,
  };

  recentEvents.push(ev);
  if (recentEvents.length > MAX_EVENTS) {
    recentEvents.splice(0, recentEvents.length - MAX_EVENTS);
  }

  if ((runtimeChannel as any).hasSubscribers) {
    try { runtimeChannel.publish(ev); } catch {}
  }
}

export function listRuntimeEvents(opts?: { limit?: number; type?: string }): RuntimeEvent[] {
  const limit = Math.max(1, Math.min(500, Math.trunc(opts?.limit ?? 100)));
  const type = opts?.type?.trim();

  const arr = type ? recentEvents.filter((e) => e.type === type) : recentEvents;
  return arr.slice(-limit);
}

export function getRuntimeStats(): RuntimeStats {
  const mem = process.memoryUsage();

  const meanMs = Number((lag as any).mean || 0) / 1e6;
  const maxMs = Number((lag as any).max || 0) / 1e6;
  const minMs = Number((lag as any).min || 0) / 1e6;
  const stddevMs = Number((lag as any).stddev || 0) / 1e6;

  const pressure: RuntimePressure =
    maxMs > 250 || mem.rss > 1_073_741_824
      ? 'hot'
      : maxMs > 80 || mem.rss > 536_870_912
        ? 'warm'
        : 'ok';

  return {
    pressure,
    eventLoop: {
      meanMs,
      maxMs,
      minMs,
      stddevMs,
    },
    memory: {
      rss: mem.rss,
      heapUsed: mem.heapUsed,
      heapTotal: mem.heapTotal,
      external: mem.external,
      arrayBuffers: mem.arrayBuffers,
    },
    uptimeSec: process.uptime(),
    recentEventCount: recentEvents.length,
  };
}

export function shouldDeferBackgroundWork(
  kind: 'prefetch' | 'watcher' | 'refresh' | 'prewarm',
): boolean {
  const s = getRuntimeStats();
  if (s.pressure === 'hot') return true;
  if (s.pressure === 'warm') return kind === 'prefetch' || kind === 'prewarm';
  return false;
}

import { createRequire } from 'node:module';
import * as path from 'node:path';
import * as fs from 'node:fs';

export interface PolicyState {
  useDdg: boolean;
  useBing: boolean;
  useYahoo: boolean;
  useMojeek: boolean;
  enableSSE: boolean;
  enableAutoRag: boolean;
  enablePrefetch: boolean;
  enableConditionalGet: boolean;
  enableRagEarlyExit: boolean;
  enableTrigramRecall: boolean;
  enableDomainCap: boolean;
  enableWatcherBackgroundFlush: boolean;

export interface PolicyContext {
  pressure?: 'ok' | 'warm' | 'hot';
  engineHealth?: Record<string, { rate?: number; drift?: boolean }>;
  selfTestCriticalFailed?: number;
}

const DEFAULT_POLICY: PolicyState = {
  useDdg: true,
  useBing: true,
  useYahoo: true,
  useMojeek: true,
  enableSSE: true,
  enableAutoRag: true,
  enablePrefetch: true,
  enableConditionalGet: true,
  enableRagEarlyExit: true,
  enableTrigramRecall: true,
  enableDomainCap: true,
  enableWatcherBackgroundFlush: true,

let DatabaseSync: any = null;
let dbAvailable = false;

try {
  const anyProc: any = globalThis.process as any;
  if (anyProc && typeof anyProc.getBuiltinModule === 'function') {
    const mod = anyProc.getBuiltinModule('node:sqlite');
    if (mod?.DatabaseSync) {
      DatabaseSync = mod.DatabaseSync;
      dbAvailable = true;
    }
  }
} catch {}

if (!dbAvailable) {
  try {
    const req = createRequire(import.meta.url);
    ({ DatabaseSync } = req('node:sqlite'));
    dbAvailable = !!DatabaseSync;
  } catch {
    dbAvailable = false;
  }
}

const DB_DIR = path.resolve(process.cwd(), 'src/audit');
const DB_PATH = path.join(DB_DIR, 'knowledge.db');

let db: any = null;
let initFailed = false;

const policyChannel = channel('vite-native-scraper.policy');

function emitPolicyEvent(type: string, payload?: unknown): void {
  if ((policyChannel as any).hasSubscribers) {
    try { policyChannel.publish({ ts: Date.now(), type, payload }); } catch {}
  }
}

function ensureDbDir() {
  if (!fs.existsSync(DB_DIR)) fs.mkdirSync(DB_DIR, { recursive: true });
}

function getDb(): any {
  if (!dbAvailable || initFailed) return null;
  if (db) return db;

  try {
    ensureDbDir();
    db = new DatabaseSync(DB_PATH);

    try { db.exec(`PRAGMA journal_mode = WAL`); } catch {}
    try { db.exec(`PRAGMA synchronous = NORMAL`); } catch {}
    try { db.exec(`PRAGMA busy_timeout = 3000`); } catch {}

    db.exec(`
      CREATE TABLE IF NOT EXISTS policy_flags (
        name TEXT PRIMARY KEY,
        valueJson TEXT NOT NULL,
        ts INTEGER NOT NULL
      ) STRICT
    `);

    return db;
  } catch {
    initFailed = true;
    return null;
  }
}

export function getPolicyOverrides(): Partial<PolicyState> {
  const conn = getDb();
  if (!conn) return {};

  try {
    const rows = conn.prepare(
      'SELECT name, valueJson FROM policy_flags ORDER BY ts DESC',
    ).all() as Array<{ name: string; valueJson: string }>;

    const out: Partial<PolicyState> = {};
    for (const row of rows) {
      try {
        (out as any)[row.name] = JSON.parse(row.valueJson);
      } catch {}
    }
    return out;
  } catch {
    return {};
  }
}

export function setPolicyOverride<K extends keyof PolicyState>(key: K, value: PolicyState[K]): boolean {
  const conn = getDb();
  if (!conn) return false;

  try {
    conn.prepare(`
      INSERT INTO policy_flags (name, valueJson, ts)
      VALUES (?, ?, ?)
      ON CONFLICT(name) DO UPDATE SET
        valueJson = excluded.valueJson,
        ts = excluded.ts
    `).run(String(key), JSON.stringify(value), Date.now());

    emitPolicyEvent('policy.override.set', { key, value });
    return true;
  } catch {
    return false;
  }
}

export function clearPolicyOverride<K extends keyof PolicyState>(key: K): boolean {
  const conn = getDb();
  if (!conn) return false;

  try {
    conn.prepare(`DELETE FROM policy_flags WHERE name = ?`).run(String(key));
    emitPolicyEvent('policy.override.clear', { key });
    return true;
  } catch {
    return false;
  }
}

export function evaluatePolicy(ctx?: PolicyContext): PolicyState {
  const base: PolicyState = {
    ...DEFAULT_POLICY,
    ...getPolicyOverrides(),
  };

  const pressure = ctx?.pressure ?? 'ok';
  const health = ctx?.engineHealth ?? {};
  const criticalFails = ctx?.selfTestCriticalFailed ?? 0;

  if (pressure === 'hot') {
    base.enablePrefetch = false;
    base.enableSSE = false;
    base.enableAutoRag = false;
    base.enableWatcherBackgroundFlush = false;
  } else if (pressure === 'warm') {
    base.enablePrefetch = false;
  }

  if (health.ddg?.drift) base.useDdg = false;
  if (health.bing?.drift) base.useBing = false;
  if (health.yahoo?.drift) base.useYahoo = false;
  if (health.mojeek?.drift) base.useMojeek = false;

  if (criticalFails > 0) {
    base.enablePrefetch = false;
    base.enableSSE = false;
    base.enableAutoRag = false;
  }

  if (!base.useDdg && !base.useBing && !base.useYahoo && !base.useMojeek) {
    base.useDdg = true;
  }

  return base;
}

export function applySelfTestRemediation(result: { checks?: Array<{ id: string; pass: boolean }> }): void {
  const checks = result?.checks || [];
  const failed = new Set(checks.filter((c) => !c.pass).map((c) => c.id));

  if (failed.has('parser.bing.href.capture') || failed.has('parser.bing.href.good')) {
    void setPolicyOverride('useBing', false);
  }
  if (failed.has('parser.mojeek.href.capture') || failed.has('parser.mojeek.href.good')) {
    void setPolicyOverride('useMojeek', false);
  }
}

export function describePolicy(ctx?: PolicyContext): {
  defaults: PolicyState;
  overrides: Partial<PolicyState>;
  effective: PolicyState;
} {
  const overrides = getPolicyOverrides();
  return {
    defaults: { ...DEFAULT_POLICY },
    overrides,
    effective: evaluatePolicy(ctx),
  };
}

export function recordReplay(
  route: string,
  request: unknown,
  response: unknown,
  statusCode = 200,
  ok = true,
  durationMs = 0,
  note?: string,
): number | null {
  const conn = getDb();
  if (!conn) return null;

  try {
    const stmt = conn.prepare(`
      INSERT INTO replay_log
      (route, requestJson, responseJson, statusCode, ok, durationMs, ts, note)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `);

    const result = stmt.run(
      route,
      JSON.stringify(request),
      JSON.stringify(response),
      statusCode,
      ok ? 1 : 0,
      Number.isFinite(durationMs) ? durationMs : 0,
      Date.now(),
      note ?? null,
    );

    try {
      const row = conn.prepare(`SELECT COUNT(*) AS c FROM replay_log`).get() as { c: number };
      if ((row?.c ?? 0) > 5000) {
        conn.prepare(`
          DELETE FROM replay_log
          WHERE id IN (
            SELECT id FROM replay_log ORDER BY ts ASC LIMIT 1000
          )
        `).run();
      }
    } catch {}

    return Number(result?.lastInsertRowid ?? 0) || null;
  } catch {
    return null;
  }
}

export function summarizeReplayDiff(
  stored: any,
  current: any,
): {
  equal: boolean;
  storedHash: string;
  currentHash: string;
  changedTopLevelKeys: string[];
  resultCount?: { stored: number; current: number };
  firstUrls?: { stored: string[]; current: string[] };
} {
  const stable = (v: any) => JSON.stringify(v);
  const hash = (s: string) =>
    crypto.createHash('sha256').update(s).digest('hex');

  const storedStr = stable(stored);
  const currentStr = stable(current);

  const storedObj = stored && typeof stored === 'object' ? stored : {};
  const currentObj = current && typeof current === 'object' ? current : {};

  const allKeys = new Set([...Object.keys(storedObj), ...Object.keys(currentObj)]);

  const changedTopLevelKeys = [...allKeys].filter((k) => {
    try {
      return JSON.stringify(storedObj[k]) !== JSON.stringify(currentObj[k]);
    } catch {
      return true;
    }
  });

  const out: ReturnType<typeof summarizeReplayDiff> = {
    equal: storedStr === currentStr,
    storedHash: hash(storedStr),
    currentHash: hash(currentStr),
    changedTopLevelKeys,
  };

  if (Array.isArray(storedObj.results) || Array.isArray(currentObj.results)) {
    out.resultCount = {
      stored: Array.isArray(storedObj.results) ? storedObj.results.length : 0,
      current: Array.isArray(currentObj.results) ? currentObj.results.length : 0,
    };

    out.firstUrls = {
      stored: Array.isArray(storedObj.results)
        ? storedObj.results.slice(0, 5).map((r: any) => r?.url).filter(Boolean)
        : [],
      current: Array.isArray(currentObj.results)
        ? currentObj.results.slice(0, 5).map((r: any) => r?.url).filter(Boolean)
        : [],
    };
  }

  return out;
}

export async function createHotBackup(label?: string): Promise<{
  available: boolean;
  ok: boolean;
  file?: string;
  sizeBytes?: number;
  sha256?: string;
  pagesCopied?: number;
  reason?: string;
}> {
  if (!dbAvailable || typeof sqliteBackup !== 'function') {
    return {
      available: false,
      ok: false,
      reason: 'node:sqlite backup API unavailable',
    };
  }

  if (!fs.existsSync(DB_PATH)) {
    return {
      available: true,
      ok: false,
      reason: 'knowledge.db does not exist',
    };
  }

  ensureSnapshotDir();

  const stamp = new Date().toISOString().replace(/[:.]/g, '-');
  const suffix = label ? `-${safeName(label)}` : '';
  const outFile = path.join(BACKUP_DIR, `knowledge-${stamp}${suffix}.db`);

  let sourceDb: any = null;
  try {
    sourceDb = new DatabaseSync(DB_PATH);

    const pagesCopied = await sqliteBackup(sourceDb, outFile, {
      rate: 32,
    });

    const st = fs.statSync(outFile);
    const hash = sha256File(outFile);

    return {
      available: true,
      ok: true,
      file: path.relative(process.cwd(), outFile).replace(/\\/g, '/'),
      sizeBytes: st.size,
      sha256: hash,
      pagesCopied,
    };
  } catch (e: any) {
    return {
      available: true,
      ok: false,
      reason: e?.message || 'backup failed',
    };
  } finally {
    closeDb(sourceDb);
  }
}

function extractFnBody(src: string, fnName: string): string {
  const candidates = [`async function ${fnName}`, `function ${fnName}`];
  let start = -1;
  for (const c of candidates) {
    start = src.indexOf(c);
    if (start >= 0) break;
  }
  if (start < 0) return '';
  const brace = src.indexOf('{', start);
  if (brace < 0) return src.slice(start, Math.min(src.length, start + 5000));
  let depth = 0;
  for (let i = brace; i < src.length; i++) {
    if (src[i] === '{') depth++;
    else if (src[i] === '}') {
      depth--;
      if (depth === 0) return src.slice(start, i + 1);
    }
  }
  return src.slice(start, Math.min(src.length, start + 8000));
}

      headers:{'User-Agent':'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36','Accept':'text/html,application/xhtml+xml,application/xml;q=0.9,text/plain;q=0.8,application/json;q=0.7,*/*;q=0.6','Accept-Language':'en-US,en;q=0.5','Accept-Encoding':acceptEncoding()},
    } as any,(res)=>{
      saw=true; const sc=res.statusCode||0; const loc=hv(res.headers.location);
      const ra=parseRetryAfterMs(hv(res.headers['retry-after']));
      if (sc>=300&&sc<400&&loc){res.resume();nodeFetch(new URL(loc,url).toString(),tMs,rCnt+1).then(ok).catch(er);return;}
      if (sc>=400){res.resume();tripBreaker(hn,sc===429||sc===503?ra:null);er(new Error(`HTTP ${sc}`));return;}
      const ct=hv(res.headers['content-type']);
      if (ct&&!SAFE_CT.some(re=>re.test(ct))){res.resume();er(new Error(`Unsupported content-type: ${ct}`));return;}
      const xRobots=hv(res.headers['x-robots-tag']).toLowerCase();
      if (xRobots&&/\bnoindex\b/.test(xRobots)&&!xRobots.includes('index,')){res.resume();er(new Error('Blocked by X-Robots-Tag: noindex'));return;}
      const chunks: Buffer[]=[]; let total=0;
      res.on('data',(c:Buffer)=>{if(settled)return;total+=c.length;if(total>MAX_COMPRESSED_BYTES){const e=new Error('Payload too large');req.destroy(e);er(e);return;}chunks.push(c);});
      res.on('end',async()=>{
        if(settled)return;
        try {
          const decoded=await decompBody(Buffer.concat(chunks),hv(res.headers['content-encoding']));
          if(looksLikeBlockPage(decoded)){tripBreaker(hn,null);er(new Error('Blocked by upstream anti-bot page'));return;}
          circuitBreaker.delete(hn); ok(decoded);
        } catch(e){er(e);}
      });
    });
    req.on('error',(e:any)=>{
      if(!saw&&ai+1<prep.addrs.length&&isRetriable(e)){reqTarget(prep,url,tMs,rCnt,ai+1).then(ok).catch(er);return;}
      er(e);
    });
    req.on('timeout',()=>{const e:any=new Error('Timeout');e.code='ETIMEDOUT';req.destroy(e);er(e);});
  });
}

function decEnt(t: string): string {

    .replace(/&#x([0-9a-f]+);/gi,(_,n)=>{try{const c=parseInt(n,16);return c>=0&&c<=0x10ffff?String.fromCodePoint(c):' ';}catch{return' ';}});
}

interface TokenBucket { tokens: number; lastRefill: number; }

function checkRateLimit(cls: string): {allowed:boolean;retryAfterSec:number} {
  const now=Date.now(); let b=rateLimiters.get(cls);
  if(!b){b={tokens:RATE_CAP,lastRefill:now};rateLimiters.set(cls,b);}

}

// Inbound origin isolation
function headerStr(req:any,name:string): string { const v=req.headers?.[name.toLowerCase()];return Array.isArray(v)?String(v[0]||''):String(v||''); }
function sameOrigin(a:string,b:string): boolean { try{const ua=new URL(a),ub=new URL(b);return ua.protocol===ub.protocol&&ua.host===ub.host;}catch{return false;} }
function rejectCrossOrigin(req:any,res:any): boolean {
  const sfs=headerStr(req,'sec-fetch-site').toLowerCase();
  if(!sfs||sfs==='same-origin'||sfs==='none') return false;
  if(sfs==='cross-site'){sendJson(res,{error:'Cross-site native API request blocked'},403);return true;}
  const self=`${headerStr(req,'x-forwarded-proto')||'http'}://${headerStr(req,'host')||'localhost'}`;
  const origin=headerStr(req,'origin'); const referer=headerStr(req,'referer');
  if(origin&&!sameOrigin(origin,self)){sendJson(res,{error:'Origin mismatch'},403);return true;}
  if(!origin&&referer&&!sameOrigin(referer,self)){sendJson(res,{error:'Referer mismatch'},403);return true;}
  return false;
}

// --- RRF + MMR ---
interface RR extends SR { rrf: number; final: number; ranks?: Record<string,number>; }

function simHashLite(t: string): string {
  const toks=t.toLowerCase().match(/[a-z0-9]+/g)||[]; if(!toks.length) return '0'.repeat(16);
  const v=new Array(64).fill(0);
  for(const tok of toks){const h=crypto.createHash('sha256').update(tok).digest();let bits=0n;for(let i=0;i<8;i++)bits=(bits<<8n)|BigInt(h[i]);for(let b=0;b<64;b++)v[b]+=((bits>>BigInt(b))&1n)===1n?1:-1;}
  let fp=0n; for(let b=0;b<64;b++) if(v[b]>0) fp|=1n<<BigInt(b);
  return fp.toString(16).padStart(16,'0');
}

function rrfFuse(lists: Record<string,SR[]>): Map<string,RR> {
  const policy=currentPolicy();
  const health=getEngineHealth();
  const map=new Map<string,RR>();
  for(const [engine,list] of Object.entries(lists)){
    const weight=health[engine]?.rate??1.0;
    const seen=new Set<string>();
    list.forEach((r,i)=>{
      if(!r.url||seen.has(r.url)) return; seen.add(r.url);
      const c=(1/(RRF_K+i+1))*weight;
      const e=map.get(r.url);
      if(e){e.rrf+=c;e.final+=c;(e.ranks??={})[engine]=i+1;if(!e.description&&r.description){e.description=r.description;e.content=r.content;}}
      else map.set(r.url,{...r,rrf:c,final:c,ranks:{[engine]:i+1}});
    });
  }
  return map;
}

function mmrDiv(items: RR[], count: number): SR[] {
  if(items.length<=count) return items.map(({rrf:_r,final:_f,...r})=>r);
  const cand=items.map(r=>({r,sh:simHashLite(`${r.title} ${r.description}`)}));
  const sel: typeof cand=[];
  const domCounts=new Map<string,number>();
  const policy=currentPolicy();
  while(sel.length<count&&cand.length){
    let bestI=0,best=-Infinity;
    for(let i=0;i<cand.length;i++){
      const c=cand[i]; const rel=c.r.final;
      // Domain diversity penalty
      let domainPenalty=1;
      if(policy.enableDomainCap){
        try{const d=new URL(c.r.url).hostname;const dc=domCounts.get(d)||0;if(dc>=2)domainPenalty=1/(1+(dc-1)*0.4);}catch{}
      }
      let maxSim=0; for(const s of sel){const d=hammingDistanceHex(c.sh,s.sh);const sim=1-d/64;if(sim>maxSim)maxSim=sim;}
      const score=MMR_LAMBDA*(rel*domainPenalty)-(1-MMR_LAMBDA)*maxSim;
      if(score>best){best=score;bestI=i;}
    }
    const picked=cand.splice(bestI,1)[0];
    sel.push(picked);
    try{const d=new URL(picked.r.url).hostname;domCounts.set(d,(domCounts.get(d)||0)+1);}catch{}
  }
  return sel.map(({r:{rrf:_r,final:_f,...rest}})=>rest);
}

// --- Fanout + metasearch ---
async function fanout(q: string, count: number): Promise<SR[]> {
  const policy=currentPolicy();
  const throttled=shouldDeferBackgroundWork('prefetch'); // proxy for heavy load
  let engines: Array<'ddg'|'bing'|'yahoo'|'mojeek'>=[];
  if(policy.useDdg) engines.push('ddg');
  if(policy.useBing) engines.push('bing');
  if(policy.useYahoo) engines.push('yahoo');
  if(policy.useMojeek) engines.push('mojeek');
  if(engines.length===0) engines=['ddg'];

  const [ddg,bing,yahoo,moj]=await Promise.all([
    engines.includes('ddg')?searchDDG(q).catch(()=>[] as SR[]):Promise.resolve([] as SR[]),
    engines.includes('bing')?searchBing(q).catch(()=>[] as SR[]):Promise.resolve([] as SR[]),
    engines.includes('yahoo')?searchYahoo(q).catch(()=>[] as SR[]):Promise.resolve([] as SR[]),
    engines.includes('mojeek')?searchMojeek(q).catch(()=>[] as SR[]):Promise.resolve([] as SR[]),
  ]);

  let all=[...ddg,...bing,...yahoo,...moj];
  // Query relaxation on low recall
  if(all.length<Math.max(3,Math.floor(count*0.3))){
    const toks=q.toLowerCase().match(/[a-z0-9]+/g)?.filter(w=>w.length>3)??[];
    if(toks.length>=3){
      const relaxed=toks.sort((a,b)=>b.length-a.length).slice(0,2).join(' ');
      if(relaxed!==q.toLowerCase()){
        const [rD,rB]=await Promise.all([searchDDG(relaxed).catch(()=>[] as SR[]),searchBing(relaxed).catch(()=>[] as SR[])]);
        all=all.concat(rD,rB);
      }
    }
  }

  const fused=rrfFuse({ddg,bing,yahoo,mojeek:moj});
  for(const r of all){if(r.url&&!fused.has(r.url))fused.set(r.url,{...r,rrf:0.001,final:0.001});}
  const sorted=[...fused.values()].sort((a,b)=>b.final-a.final);
  return mmrDiv(sorted,count);
}

function writeCache(key: string, results: SR[]) {
  const now=Date.now();
  if(searchCache.size>=SEARCH_CACHE_MAX){const o=searchCache.keys().next().value;if(o!==undefined)searchCache.delete(o);}
  searchCache.set(key,{results,fresh:now+SEARCH_CACHE_TTL_MS,stale:now+SEARCH_CACHE_TTL_MS+SEARCH_CACHE_SWR_MS});
  try{setCachedSearch(key,JSON.stringify(results),now+SEARCH_CACHE_TTL_MS);}catch{}
}

function bgRefresh(q: string,count: number,key: string){
  if(refreshInFlight.has(key)) return;
  refreshInFlight.add(key);
  fanout(q,count).then(r=>writeCache(key,r)).catch(()=>{}).finally(()=>refreshInFlight.delete(key));
}

async function metasearch(q: string,count: number): Promise<{results:SR[];cs:string}>{
  const sq=normQ(q), sc=clamp(count), key=`${sq.toLowerCase()}::${sc}`, now=Date.now();
  // L1 with XFetch probabilistic early refresh
  const l1=searchCache.get(key);
  if(l1){
    if(now<l1.fresh){
      const delta=SEARCH_CACHE_TTL_MS/4;
      const roll=-delta*0.5*Math.log(Math.max(Math.random(),1e-10));
      searchCache.delete(key);searchCache.set(key,l1);
      if(now+roll>=l1.fresh)queueMicrotask(()=>bgRefresh(sq,sc,key));
      return{results:l1.results,cs:'fresh'};
    }
    if(now<l1.stale){searchCache.delete(key);searchCache.set(key,l1);bgRefresh(sq,sc,key);return{results:l1.results,cs:'stale'};}
  }
  // L2
  try{const l2=getCachedSearch(key);if(l2&&now<l2.expires){const r=JSON.parse(l2.payload) as SR[];searchCache.set(key,{results:r,fresh:l2.expires,stale:l2.expires+SEARCH_CACHE_SWR_MS});return{results:r,cs:'l2'};}}catch{}
  // In-flight coalescing
  const existing=inflightSearches.get(key);
  if(existing) return existing;
  const flight=(async():Promise<{results:SR[];cs:string}>=>{
    try{const r=await fanout(sq,sc);writeCache(key,r);return{results:r,cs:'miss'};}
    finally{inflightSearches.delete(key);}
  })();
  inflightSearches.set(key,flight);
  return flight;
}

  const st=await fs.promises.stat(rp);
  if(!st.isFile()) throw new Error('Must be a file');
  if(st.size>MAX_INGEST_BYTES) throw new Error('Too large');
  const content=await fs.promises.readFile(rp,'utf-8');
  const ih=sha256T(content);
  const isCode=/\.(ts|tsx|js|jsx|py|go|rs|java|css|json|md)$/i.test(tp);
  const safe=scrubForPersistence(isCode?content.slice(0,1200).replace(/\/\*[\s\S]*?\*\/|\/\/.*/g,' ').replace(/\s+/g,' ').trim().slice(0,500):stripHtml(content).slice(0,500));
  let chunks=0;
  try{purgeKnowledge(tp);chunks=recordKnowledgeChunked('ingest',tp,scrubForPersistence(content));}catch{}
  try{recordCodeEdges(tp,content);}catch{}
  const ah=await appendAudit({type:'ingest',path:tp,ingestedHash:ih,summaryLen:safe.length,isCode,chunks,timestamp:Date.now()});
  broadcastEvent('ingest.complete',{path:tp,chunks});
  emitRuntimeEvent('ingest.done',{path:tp,chunks});
  return{success:true,auditHash:ah,summary:safe,ingestedHash:ih,chunks};
}

// --- Maintenance stats ---
function getMaintenanceStats(): any {
  try{
    const conn=(globalThis as any).__knowledgeDbRef;
    if(!conn) return{available:false};
    const pageSize=(conn.prepare?.('PRAGMA page_size').get() as any)?.page_size??4096;
    const pageCount=(conn.prepare?.('PRAGMA page_count').get() as any)?.page_count??0;
    const freelistCount=(conn.prepare?.('PRAGMA freelist_count').get() as any)?.freelist_count??0;
    return{available:true,pageSize,pageCount,freelistCount,approxDbBytes:pageSize*pageCount,reclaimableBytes:pageSize*freelistCount};
  }catch{return{available:false};}
}

// --- Auto-RAG watcher ---
function startAutoRag(srcRoot: string): ()=>void {
  const debounce=new Map<string,ReturnType<typeof setTimeout>>();
  const pendingBatch=new Map<string,string>();
  let batchTimer: ReturnType<typeof setTimeout>|null=null;
  let watcher: fs.FSWatcher|null=null;

  const flushBatch=async()=>{
    if(!pendingBatch.size) return;
    const snap=new Map(pendingBatch); pendingBatch.clear(); batchTimer=null;
    for(const [ref,content] of snap){
      try{
        if(isIngestDenied(ref)){broadcastEvent('watcher.skip',{ref,reason:'denylist'});continue;}
        if(shouldSkipIngest(ref,content)){broadcastEvent('watcher.skip',{ref,reason:'content-hash-unchanged'});continue;}
        if(isNearDuplicateContent(ref,content,2)){broadcastEvent('watcher.skip',{ref,reason:'simhash-near-dup'});continue;}
        purgeKnowledge(ref);
        const n=recordKnowledgeChunked('ingest',ref,scrubForPersistence(content));
        try{recordCodeEdges(ref,content);}catch{}
        broadcastEvent('watcher.ingest',{ref,chunks:n,ts:Date.now()});
        emitRuntimeEvent('watcher.ingest',{ref,chunks:n});
      }catch{}
    }
    try{maybeOptimizeDb();}catch{}
  };

  try{
    watcher=fs.watch(srcRoot,{recursive:true},(_ev,relName)=>{
      if(!relName) return;
      const fp=path.join(srcRoot,relName);
      if(fp.includes(path.sep+'audit'+path.sep)) return;
      if(!WATCHER_EXTS.test(relName)) return;
      const existing=debounce.get(fp);if(existing) clearTimeout(existing);
      debounce.set(fp,setTimeout(async()=>{
        debounce.delete(fp);
        try{
          const st=await fs.promises.stat(fp).catch(()=>null);
          if(!st||!st.isFile()||st.size>MAX_INGEST_BYTES) return;
          const content=await fs.promises.readFile(fp,'utf-8');
          const rel=path.relative(process.cwd(),fp).replace(/\\/g,'/');
          pendingBatch.set(rel,content);
          if(batchTimer) clearTimeout(batchTimer);
          if(pendingBatch.size>=20) void flushBatch();
          else batchTimer=setTimeout(()=>{void flushBatch();},500);
        }catch{}
      },WATCHER_DEBOUNCE_MS));
    });
  }catch(e){console.warn('[Auto-RAG] fs.watch failed:',(e as any)?.message);}

  return ()=>{
    if(watcher){try{watcher.close();}catch{}}
    for(const t of debounce.values()) clearTimeout(t);
    if(batchTimer) clearTimeout(batchTimer);
    debounce.clear(); pendingBatch.clear();
  };
}

    name:'vite-native-scraper',

      },8000).unref?.();
    },
    configurePreviewServer(server){ attachRoutes(server.middlewares); },
  };
}
